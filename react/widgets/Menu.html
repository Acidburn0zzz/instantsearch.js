                                                  Description

                                                  The Menu component displays a menu that lets the user choose a single value for a specific attribute.

                                                  See live example

                                                  Props

                                                  attributeName*
                                                  		type: string 

                                                  the name of the attribute in the record

                                                  showMore
                                                  		type: booleandefault: false

                                                  true if the component should display a button that will expand the number of items

                                                  limitMin
                                                  		type: numberdefault: 10

                                                  the minimum number of diplayed items

                                                  limitMax
                                                  		type: numberdefault: 20

                                                  the maximun number of displayed items. Only used when showMore is set to true

                                                  defaultRefinement
                                                  		type: string 

                                                  the value of the item selected by default

                                                  transformItems
                                                  		type: function 

                                                  Function to modify the items being displayed, e.g. for filtering or sorting them. Takes an items as parameter and expects it back in return.

                                                  withSearchBox
                                                  		type: booleandefault: false

                                                  true if the component should display an input to search for facet values

                                                  Requirements

                                                  The attribute passed to the attributeName prop must be present in "attributes for faceting" on the Algolia dashboard or configured as attributesForFaceting via a set settings call to the Algolia API.

                                                  Example usage

                                                    
                                                  Code
                                                  
  
                                                  
    import React from 'react';

import {Menu, InstantSearch} from '../packages/react-instantsearch/dom';

export default function App() {
  return (
    <InstantSearch
      appId="latency"
      apiKey="6be0576ff61c053d5f9a3225e2a90f76"
      indexName="ikea"
    >
      <Menu
        attributeName="category"
      />
    </InstantSearch>
  );
}

                                                  CSS classes

                                                  .ais-Menu__root {}

                                                  the root of the component

                                                  .ais-Menu__items {}

                                                  the container of all items in the menu

                                                  .ais-Menu__item {}

                                                  a single item

                                                  .ais-Menu__itemLinkSelected {}

                                                  the selected menu item

                                                  .ais-Menu__itemLink {}

                                                  the item link

                                                  .ais-Menu__itemLabelSelected {}

                                                  the selected item label

                                                  .ais-Menu__itemLabel {}

                                                  the item label

                                                  .ais-Menu__itemCount {}

                                                  the item count

                                                  .ais-Menu__itemCountSelected {}

                                                  the selected item count

                                                  .ais-Menu__noRefinement {}

                                                  present when there is no refinement

                                                  .ais-Menu__showMore {}

                                                  the button that let the user toggle more results

                                                  .ais-Menu__SearchBox {}

                                                  the container of the search for facet values searchbox. See the SearchBox documentation for the classnames and translation keys of the SearchBox.

                                                  Translation keys

                                                  showMore

                                                  The label of the show more button. Accepts one parameters, a boolean that is true if the values are expanded

                                                  noResults

                                                  The label of the no results text when no search for facet values results are found.
